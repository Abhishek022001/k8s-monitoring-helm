{{ define "agent.config.node_exporter" }}
// Node Exporter
discovery.relabel "node_exporter" {
  targets = discovery.kubernetes.services.targets
{{- if .Values.metrics.node_exporter.deploy }}
  rule {
    source_labels = ["__meta_kubernetes_service_label_app_kubernetes_io_instance"]
    regex = "{{ .Release.Name }}"
    action = "keep"
  }
{{- end }}
  rule {
    source_labels = ["__meta_kubernetes_service_label_app_kubernetes_io_name"]
    regex = "prometheus-node-exporter"
    action = "keep"
  }
  rule {
    source_labels = ["__meta_kubernetes_service_port_name"]
    regex = "metrics"
    action = "keep"
  }
}

prometheus.scrape "node_exporter" {
  job_name   = "integrations/node_exporter"
  targets  = discovery.relabel.node_exporter.output
  forward_to = [prometheus.relabel.node_exporter.receiver]
{{- if .Values.metrics.node_exporter.service.isTLS }}
  scheme     = "https"
  bearer_token_file = "/var/run/secrets/kubernetes.io/serviceaccount/token"
  tls_config {
    insecure_skip_verify = true
  }
{{- end }}
}

prometheus.relabel "node_exporter" {
  rule {
    source_labels = ["__name__"]
    regex = "up|{{ join "|" .Values.metrics.node_exporter.allowList }}"
    action = "keep"
  }
  forward_to = [prometheus.relabel.add_cluster_label.receiver]
}
{{ end }}
